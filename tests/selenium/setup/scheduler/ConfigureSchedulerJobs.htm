Configure Scheduler Jobs
open /config/scheduler
click link=Inbound Email Message Processor
click css=#frmJobcron_parser > label > input[name="enabled"]
type css=#frmJobcron_parser > input[name="max_messages"] 25
click css=#frmJobcron_parser > button.submit
click link=Maintenance
click css=#frmJobcron_maint > label > input[name="enabled"]
type css=#frmJobcron_maint > input[name="duration"] 1
select css=#frmJobcron_maint > select[name="term"] label=day(s)
type css=#frmJobcron_maint > input[name="starting"] midnight
click css=#frmJobcron_maint > button.submit
pause 500
